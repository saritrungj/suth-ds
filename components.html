Modal — Delete confirmation & success
Confirm delete, then show success on complete (no toast noise) 
<!-- Delete confirmation -->
<div class="modal-overlay" id="modalDeleteConfirm">
  <div class="modal modal-alert modal-sm" role="dialog" aria-modal="true" aria-labelledby="deleteTitle">
    <div class="modal-alert-icon error" aria-hidden="true">
      <svg width="28" height="28" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5">
        <polyline points="3 6 5 6 21 6"/>
        <path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"/>
      </svg>
    </div>
    <div class="modal-header">
      <h3 class="modal-title" id="deleteTitle">Delete record?</h3>
    </div>
    <div class="modal-body">
      Are you sure you want to delete this record? This action cannot be undone.
    </div>
    <div class="modal-footer">
      <button type="button" class="btn btn-secondary" onclick="SUTHDS.modal.close('modalDeleteConfirm')">Cancel</button>
      <button type="button" class="btn btn-danger" onclick="completeDelete()">Delete</button>
    </div>
  </div>
</div>

<!-- Success on complete -->
